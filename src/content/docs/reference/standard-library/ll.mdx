import ll_categories from '~/data/ll_categories.json';
import lsl_definitions from '~/data/lsl_definitions.json';
import List from '@components/List.jsx';

<List items={Object.entries(ll_categories).filter(([key, _]) => {
    let functionName = key.replace('ll.', 'll');
    let definition = lsl_definitions.functions[functionName];
    if(!definition) return true; // Doesn't exist in LSL?
    if(definition.private) return false; // Do not include in documentation
    if(definition.deprecated) return false; // Do not include deprecated functions
    if([
        'llSetTimerEvent',
        'llResetTime',
        'llGetAndResetTime',
        'llSetMemoryLimit',
    ].includes(functionName)) return false; // Functions removed from ll.* specifically
    return true;
})} client:load/>

## Removed
:::danger[Removed]
The following functions are not available in `ll` and may be available in `llcompat` instead:
```slua
function ll.SetTimerEvent(seconds: number): void
function ll.ResetTime(): void
function ll.GetAndResetTime(): number
function ll.SetMemoryLimit(limit: number): number
```
:::
